#include<stdio.h>
void main()
{
 char str[10],str2[10];
 int i,j,n,count=0;
 printf("Enter the string: ");
 scanf("%s",str);
 for(i=0;str[i]!='\0';i++)
 {
   n=count++;
 }
  printf("Enter the second string: ");
  scanf("%s",str2);
  count=0;
  for(i=0;i<n;i++)
  {
   for(j=0;str2[j]!='\0';j++)
   {
     if(str[i]==str2[j])
      	count++;
   }
  }
  if(n==count)
  {
   printf("ANAGRAM");
  }
  else
  {
   printf("NOT ANAGRAM");
  }
 
}
