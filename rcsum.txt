#include<stdio.h>
void main()
{
	int a[100][100];
	int i,j,r,c,rc,sum; 
	printf("Enter how many rows \n");
	scanf("%d",&r);
	printf("Enter how many cols \n");
	scanf("%d",&c);
	printf("Enter elements into array \n");
	for(i=1;i<=r;i++)
	{
		for(j=1;j<=c;j++)
		{
			scanf("%d",&a[i][j]);
		}
	}
	printf("Enter the column number to print its sum:");
      scanf("%d",&rc);
	for(i=1;i<=c;i++)
	{
        if(rc==i)
        {
		sum=0;
		for(j=1;j<=r;j++)
		{
			sum=sum+a[j][i]	;
		}
        }
	}
	printf(" %d Colm sum= %d",rc,sum);		
	
}
