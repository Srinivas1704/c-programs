#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main() 
{
    char str[1000];
    int i,x[10]={0};
    scanf("%s",str);
    for(i=0;str[i]!='\0';i++)
    {
        if(str[i]>='0'&&str[i<=9])
        {
            int val=str[i]-'0';
            x[val]=x[val]+1;
        }
    }
    for(i=0;i<10;i++)
        printf("%d ",x[i]);
    return 0;
}
