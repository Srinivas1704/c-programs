#include<stdio.h>
void subset(char str[],int n)
{
   int x,i,k,j;
   for(i=1;i<=n;i++)
   {
     for(j=0;j<=n-i;j++)
     {
       x=j+i;
       for(k=j;k<x;k++)
       {
        printf("%3c",str[k]);
       }
      printf("\n");
     }
   }
 
}

int main()
{
 char str[10];
 scanf("%s",str);
 subset(str,strlen(str));
 return 0;
}
